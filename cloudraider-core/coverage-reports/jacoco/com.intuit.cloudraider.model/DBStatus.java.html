package com.intuit.cloudraider.model;

/**
 * Object for storing the database id and status.
 * <p>
  */
public class DBStatus {
    final private String dbInstanceId;
    final private String status;

    /**
     * Instantiates a new Db status.
     *
     * @param dbInstanceId the db instance id
     * @param status       the status
     */
    public DBStatus(String dbInstanceId, String status) {
        this.dbInstanceId = dbInstanceId;
        this.status = status;
    }

    /**
     * Gets db instance id.
     *
     * @return the db instance id
     */
    public String getDbInstanceId() {
        return dbInstanceId;
    }

    /**
     * Gets status.
     *
     * @return the status
     */
    public String getStatus() {
        return status;
    }

    @Override
    public String toString() {
        return "DBStatus{" +
                "dbInstanceId='" + dbInstanceId + '\'' +
                ", status='" + status + '\'' +
                '}';
    }
}
