package com.intuit.cloudraider.model;

/**
 * Possible CloudWatch alarm state values.
 */
public enum AlarmStateValue {
    /**
     * Ok alarm state value.
     */
    OK,
    /**
     * Alarm alarm state value.
     */
    ALARM,
    /**
     * Non existent alarm state value.
     */
    NON_EXISTENT;
}
