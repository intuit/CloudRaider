package com.intuit.cloudraider.model;

/**
 * Possible actions that can be run through AWS SSM on a particular EC2 instance.
 */
public enum Command {

    /**
     * Spikecpu command.
     */
    SPIKECPU("spikecpuCommand"),

    /**
     * Corruptnetwork command.
     */
    CORRUPTNETWORK("networkcorruptionCommand"),
    /**
     * Delaynetwork command.
     */
    DELAYNETWORK("networklatencyCommand"),
    /**
     * Packetloss command.
     */
    PACKETLOSS("networkpacketlossCommand"),
    /**
     * Killprocess command.
     */
    KILLPROCESS("killCommand"),
    /**
     * Diskfull command.
     */
    DISKFULL("diskfullCommand"),
    /**
     * Ramdiskfull command.
     */
    RAMDISKFULL("ramdiskfullCommand"),
    /**
     * Stopservice command.
     */
    STOPSERVICE("stopserviceCommand"),
    /**
     * Startservice command.
     */
    STARTSERVICE("startserviceCommand"),
    /**
     * Blockdomain command.
     */
    BLOCKDOMAIN("blockdomainCommand"),
    /**
     * Unblockdomain command.
     */
    UNBLOCKDOMAIN("unblockdomainCommand"),
    /**
     * Blockdynamodb command.
     */
    BLOCKDYNAMODB("blockDynamodbCommand"),
    /**
     * Blockport command.
     */
    BLOCKPORT("blockportCommand"),
    /**
     * Blockoutboundport command.
     */
    BLOCKOUTBOUNDPORT("blockoutboundportCommand"),
    /**
     * Unblockport command.
     */
    UNBLOCKPORT("unblockportCommand"),
    /**
     * Unblockdynamodb command.
     */
    UNBLOCKDYNAMODB("unblockDynamodbCommand"),
    /**
     * Renamefile command.
     */
    RENAMEFILE("renamefileCommand"),
    /**
     * Blocks 3 command.
     */
    BLOCKS3("blockS3Command"),
    /**
     * Unblocks 3 command.
     */
    UNBLOCKS3("unblockS3Command");


    private final String commandString;

    private Command(String commandString) {
        this.commandString = commandString;
    }

    /**
     * Gets command name.
     *
     * @return the command name
     */
    public String getCommandName() {
        return this.commandString;
    }
}
